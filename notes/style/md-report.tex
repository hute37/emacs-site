\renewcommand{\maketitle}{
  \begin{titlepage}

    \thispagestyle{firstpage}
    \addtolength{\headsep}{-20pt}

    % Top horizontal bar (theme color) 3pt - full width
    \noindent\textcolor{themecolor}{\rule{\textwidth}{3pt}}
    \vspace{-3mm}
    
    % Journal information - left-aligned, 8.5-9pt ARTICLE label
    \noindent{\fontsize{8.5}{10.2}\selectfont\sffamily\bfseries\textcolor{darkgray}{\MakeUppercase{\@articletype}}}%
    \ifdefempty{\@category}{}{%
        \quad|\quad{\fontsize{8.5}{10.2}\selectfont\sffamily\textcolor{darkgray}{\MakeUppercase{\@category}}}%
        \par
      }
      \vspace{5mm}
      
      % Article title - left-aligned, 21pt
      \noindent{\fontsize{21}{35}\selectfont\bfseries\sffamily\textcolor{themecolor}\@title}\par
      \vspace{4mm}
      
      % Article subtitle - left-aligned, 16pt
      \noindent{\fontsize{16}{35}\selectfont\bfseries\sffamily\textcolor{darkgray}\@subtitle}\par
      \vspace{4mm}
      
      % Author - left-aligned, 10pt (name bolding controlled in tex file)
      \noindent{\fontsize{9}{13.2}\selectfont\sffamily\@author}\par
      \vspace{4mm}
      
      % % Bottom horizontal bar
      % \vspace{3mm}
      % \noindent\textcolor{lightgray}{\rule{\textwidth}{0.5pt}}
      % \vspace{-1mm}
      \vspace{3mm}
        
      \begin{flushleft}
        \fontsize{8.5}{9.2}\selectfont\sffamily\color{darkgray}
        \begin{tabular}{ | >{\bfseries}r l | }
          \hline
          Project: & \@project\\
          Document: & \@docname \\
          URL: & \href{\@docsurl}{\@docsurl} \\
          Date: & \@date \\
          \hline
        \end{tabular}
      \end{flushleft}
        
      \vspace{2mm}
      % Keywords - full-width single column
      \ifdefempty{\@keywords}{}{%
          \noindent{\fontsize{9}{10.35}\selectfont\sffamily\textbf{Keywords:} \@keywords}\par
        }
            
        % Bottom horizontal bar
        \vspace{-4mm}
        \noindent\textcolor{lightgray}{\rule{\textwidth}{0.5pt}}
        \vspace{1mm}
            
        % Abstract - full-width single column
        \ifdefempty{\@abstract}{}{%

            {\fontsize{9}{10.35}\selectfont\@abstract\par}
            \vspace{4mm}
          }
          
            % Bottom horizontal bar

          \vspace{5mm}
    
    \vfill
  \end{titlepage}

}
